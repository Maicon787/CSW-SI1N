import pandas as pd
        import numpy as np
        
        # Função para remover outliers com base no Z-Score
        def remove_outlier(df_old, threshold=3):
            # Calcula a média e o desvio padrão de cada coluna numérica
            mean = df_old.mean()
            std_dev = df_old.std()
            
            # Calcula o Z-Score de cada valor
            z_scores = (df_old - mean) / std_dev
            
            # Filtra os dados para remover outliers
            df_clean = df_old[(z_scores.abs() <= threshold).all(axis=1)]
            
            return df_clean
        
        # Exemplo de uso
        # df_old é o seu DataFrame original
        df_cleaned = remove_outlier(df_old)
        
        # Exibe o DataFrame sem os outliers
        print(df_cleaned)
